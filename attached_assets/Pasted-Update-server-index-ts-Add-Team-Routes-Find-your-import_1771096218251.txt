B. Update client/src/App.tsx - Initialize PWA
Add to your component:
typescriptimport { useEffect } from 'react';
import { pwaManager } from './utils/pwa-manager';
import { PWAInstallButton } from './components/pwa-install-button';
import './styles/mobile.css';
import { MobileOptimizationProvider } from './components/mobile-optimization';

export function App() {
  useEffect(() => {
    // Initialize PWA
    pwaManager.registerServiceWorker();
  }, []);

  return (
    <MobileOptimizationProvider>
      <div className="app">
        {/* Add this button to your header */}
        <PWAInstallButton />
        
        {/* Rest of your app */}
      </div>
    </MobileOptimizationProvider>
  );
}
